"""
Networked System Reliability Analysis
======================================
15-component system modeled with Continuous-Time Markov Chains (CTMC)
Covers:
  - Monte Carlo simulation (10^5+ runs)
  - Availability, failure frequency, MTTF estimation
  - Series, parallel, k-out-of-n system reliability
  - Closed-form analytical validation
  - Sensitivity analysis on λ and μ
  - Importance sampling for rare-event estimation
  - Redundancy allocation under cost constraints
"""

import numpy as np
import pandas as pd
from scipy.linalg import expm
from scipy.optimize import minimize
import warnings
import matplotlib.pyplot as plt
from matplotlib.patches import Circle, FancyArrowPatch, FancyBboxPatch
warnings.filterwarnings("ignore")

# ─────────────────────────────────────────────
# 1. SYSTEM CONFIGURATION
# ─────────────────────────────────────────────

N_COMPONENTS = 15
N_SIMULATIONS = 100_000
T_MISSION = 1000.0       # mission time (hours)
RANDOM_SEED = 42
rng = np.random.default_rng(RANDOM_SEED)

# Real-world-inspired failure & repair rates (per hour)
# Source profile: industrial electronic/mechanical components
LAMBDA = np.array([
    0.002, 0.003, 0.001, 0.004, 0.002,
    0.005, 0.003, 0.002, 0.001, 0.004,
    0.003, 0.002, 0.005, 0.001, 0.003
])  # failure rates λ_i

MU = np.array([
    0.10, 0.08, 0.12, 0.09, 0.11,
    0.07, 0.10, 0.09, 0.13, 0.08,
    0.10, 0.11, 0.07, 0.12, 0.09
])  # repair rates μ_i


# ─────────────────────────────────────────────
# 2. ANALYTICAL STEADY-STATE (per component)
# ─────────────────────────────────────────────

def component_availability(lam, mu):
    """Steady-state availability: A = μ / (λ + μ)"""
    return mu / (lam + mu)

def component_mttf(lam):
    """Mean Time To Failure for a single component: 1/λ"""
    return 1.0 / lam

def component_mttr(mu):
    """Mean Time To Repair: 1/μ"""
    return 1.0 / mu


# ─────────────────────────────────────────────
# 3. SYSTEM TOPOLOGY RELIABILITY (Analytical)
# ─────────────────────────────────────────────

def series_reliability(availabilities):
    """All components must work: R_s = Π A_i"""
    return np.prod(availabilities)

def parallel_reliability(availabilities):
    """At least one must work: R_p = 1 - Π(1 - A_i)"""
    return 1.0 - np.prod(1.0 - availabilities)

def k_out_of_n_reliability(availabilities, k):
    """
    Exactly k-out-of-n components must work.
    Assumes homogeneous p = mean(availabilities) for tractability.
    """
    from math import comb
    n = len(availabilities)
    p = np.mean(availabilities)
    q = 1.0 - p
    return sum(comb(n, i) * (p**i) * (q**(n-i)) for i in range(k, n+1))


# ─────────────────────────────────────────────
# 4. CTMC GENERATOR MATRIX (2-state per component)
# ─────────────────────────────────────────────

def build_generator(lam, mu):
    """
    For a single component:
      State 0 = Working, State 1 = Failed
      Q = [[-λ,  λ],
           [ μ, -μ]]
    """
    Q = np.array([[-lam, lam],
                  [ mu, -mu]])
    return Q

def ctmc_availability(lam, mu, t):
    """
    Time-dependent availability via matrix exponential:
      P(t) = exp(Q*t), starting in state 0 (working)
    """
    Q = build_generator(lam, mu)
    Pt = expm(Q * t)
    return Pt[0, 0]   # P(working at t | working at 0)


# ─────────────────────────────────────────────
# 5. MONTE CARLO SIMULATION
# ─────────────────────────────────────────────

def simulate_component(lam, mu, T, rng):
    """
    Simulate one component over [0, T] using competing exponentials.
    Returns: (time_working, n_failures)
    """
    t = 0.0
    state = 1  # 1=working, 0=failed
    time_working = 0.0
    n_failures = 0

    while t < T:
        if state == 1:
            dt = rng.exponential(1.0 / lam)
            dt = min(dt, T - t)
            time_working += dt
            t += dt
            if t < T:
                state = 0
                n_failures += 1
        else:
            dt = rng.exponential(1.0 / mu)
            t += dt
            if t < T:
                state = 1

    return time_working, n_failures


def run_monte_carlo(LAMBDA, MU, T, n_sims, k_threshold=8, rng=None):
    """
    Run n_sims Monte Carlo trials for the 15-component system.
    Records per-trial: availability, failure counts, system-level status.
    """
    if rng is None:
        rng = np.random.default_rng(42)

    n = len(LAMBDA)
    availabilities = np.zeros(n_sims)
    failure_counts = np.zeros((n_sims, n), dtype=int)
    system_up_time = np.zeros(n_sims)

    for sim in range(n_sims):
        comp_avail = np.zeros(n)
        comp_failures = np.zeros(n, dtype=int)
        for i in range(n):
            tw, nf = simulate_component(LAMBDA[i], MU[i], T, rng)
            comp_avail[i] = tw / T
            comp_failures[i] = nf
        failure_counts[sim] = comp_failures
        # System availability: k-out-of-n (k=8 of 15 must work)
        n_working = np.sum(comp_avail > 0.5)
        availabilities[sim] = np.mean(comp_avail)
        system_up_time[sim] = 1.0 if n_working >= k_threshold else 0.0

    return {
        "mean_availability":      np.mean(availabilities),
        "std_availability":       np.std(availabilities),
        "system_availability":    np.mean(system_up_time),
        "mean_failures_per_comp": np.mean(failure_counts, axis=0),
        "total_failures":         failure_counts.sum(axis=1),
        "mttf_estimate":          T / (failure_counts.sum(axis=1).mean() + 1e-9),
    }


# ─────────────────────────────────────────────
# 6. SENSITIVITY ANALYSIS
# ─────────────────────────────────────────────

def sensitivity_analysis(LAMBDA, MU, perturbation=0.10):
    """
    Analytical sensitivity: perturb each λ_i and μ_i and measure change
    in system-level k-out-of-n reliability using closed-form availability.
    No Monte Carlo noise — fully deterministic.
    """
    avail_base = component_availability(LAMBDA, MU)
    baseline = k_out_of_n_reliability(avail_base, k=8)
    results = []

    for i in range(len(LAMBDA)):
        # Perturb λ_i upward (higher failure rate → lower availability)
        L_up = LAMBDA.copy(); L_up[i] *= (1 + perturbation)
        A_lam = component_availability(L_up, MU)
        sys_lam = k_out_of_n_reliability(A_lam, k=8)
        lam_sens = (sys_lam - baseline) / (perturbation * LAMBDA[i])

        # Perturb μ_i upward (faster repair → higher availability)
        M_up = MU.copy(); M_up[i] *= (1 + perturbation)
        A_mu = component_availability(LAMBDA, M_up)
        sys_mu = k_out_of_n_reliability(A_mu, k=8)
        mu_sens = (sys_mu - baseline) / (perturbation * MU[i])

        results.append({
            "Component": i + 1,
            "λ_sensitivity": round(lam_sens, 6),
            "μ_sensitivity": round(mu_sens, 6),
            "baseline_avail": round(component_availability(LAMBDA[i], MU[i]), 4),
        })

    return pd.DataFrame(results)


# ─────────────────────────────────────────────
# 7. IMPORTANCE SAMPLING (Rare Event Estimation)
# ─────────────────────────────────────────────

def importance_sampling_rare_failure(LAMBDA, MU, T, n_sims=50_000,
                                     failure_threshold=5, rng=None):
    """
    Estimate P(≥ failure_threshold components fail at least once during [0,T])
    using importance sampling with inflated Poisson failure counts.

    For each component, the number of failures in [0,T] under the true
    distribution is Poisson(λ_i·T). Under the IS distribution it is
    Poisson(λ_IS_i·T). The likelihood ratio per component is:

        w_i = Pois(n_i; λ_i·T) / Pois(n_i; λ_IS_i·T)
            = exp((λ_IS_i - λ_i)·T) · (λ_i/λ_IS_i)^n_i
    """
    if rng is None:
        rng = np.random.default_rng(42)

    beta = 5.0
    LAMBDA_IS = LAMBDA * beta
    rate_true = LAMBDA * T
    rate_IS   = LAMBDA_IS * T

    estimates = []
    for _ in range(n_sims):
        # Sample failure counts under IS distribution
        n_fail = rng.poisson(rate_IS)           # shape (N,)
        # Indicator: ≥ failure_threshold components experienced ≥1 failure
        n_failed_comps = np.sum(n_fail >= 1)
        indicator = float(n_failed_comps >= failure_threshold)
        # Log likelihood ratio (sum over components)
        log_w = np.sum((rate_IS - rate_true) + n_fail * np.log(rate_true / rate_IS))
        w = np.exp(np.clip(log_w, -40, 20))
        estimates.append(indicator * w)

    estimates = np.array(estimates)
    p_est = float(np.mean(estimates))
    se    = float(np.std(estimates) / np.sqrt(n_sims))

    return {
        "P_rare_failure_IS": p_est,
        "standard_error":    se,
        "95pct_CI":          (max(0.0, p_est - 1.96 * se), p_est + 1.96 * se),
        "target":            f"P(≥{failure_threshold} components fail ≥1 time in {T}hr)",
    }


# ─────────────────────────────────────────────
# 8. REDUNDANCY ALLOCATION UNDER COST CONSTRAINTS
# ─────────────────────────────────────────────

COMPONENT_COSTS = np.array([
    500, 800, 300, 1000, 600,
    900, 700, 500, 400,  800,
    600, 500, 900, 350,  700
])  # unit cost per additional redundant component ($)

TOTAL_BUDGET = 5000  # total redundancy budget ($)

def system_reliability_with_redundancy(redundancy, LAMBDA, MU):
    """
    Each component i has (1 + redundancy[i]) parallel copies.
    System reliability = series of parallel subsystems.
    """
    avail = component_availability(LAMBDA, MU)
    subsystem_reliabilities = []
    for i, r in enumerate(redundancy):
        r = int(round(r))
        n_copies = 1 + r
        # Parallel reliability for n_copies of component i
        sub_r = 1.0 - (1.0 - avail[i]) ** n_copies
        subsystem_reliabilities.append(sub_r)
    return np.prod(subsystem_reliabilities)   # series of subsystems

def optimize_redundancy(LAMBDA, MU, costs, budget, max_redundancy=3):
    """
    Maximize system reliability subject to budget constraint.
    Uses a greedy marginal-gain algorithm suitable for integer allocation.
    At each step: add one redundant unit to the component with highest
    reliability gain per dollar, until budget exhausted.
    """
    n = len(LAMBDA)
    avail = component_availability(LAMBDA, MU)

    def subsystem_r(i, r):
        """Parallel reliability of (1+r) copies of component i."""
        return 1.0 - (1.0 - avail[i]) ** (1 + r)

    def system_r(redun):
        return np.prod([subsystem_r(i, r) for i, r in enumerate(redun)])

    redun = np.zeros(n, dtype=int)
    remaining = budget
    baseline = system_r(redun)

    for _ in range(budget):   # at most budget/min_cost iterations
        best_i, best_gain_per_dollar = -1, 0.0
        for i in range(n):
            if redun[i] >= max_redundancy:
                continue
            if costs[i] > remaining:
                continue
            # Marginal gain from adding one unit to component i
            gain = (subsystem_r(i, redun[i] + 1) - subsystem_r(i, redun[i]))
            gain_per_dollar = gain / costs[i]
            if gain_per_dollar > best_gain_per_dollar:
                best_gain_per_dollar = gain_per_dollar
                best_i = i
        if best_i < 0:
            break
        redun[best_i] += 1
        remaining -= costs[best_i]

    return {
        "optimal_redundancy":     redun,
        "total_cost":             int(budget - remaining),
        "baseline_reliability":   round(baseline, 8),
        "optimized_reliability":  round(system_r(redun), 8),
    }

def plot_component_availability_curves(LAMBDA, MU, T=1000.0,
                                       save_path="ctmc_availability_curves.png"):
    """
    Plots time-dependent CTMC availability curves for all components.

    A_i(t) = μ_i/(λ_i + μ_i) + λ_i/(λ_i + μ_i) exp[-(λ_i + μ_i)t]
    """

    time_grid = np.linspace(0, T, 1000)

    plt.figure(figsize=(11, 6))

    for i, (lam, mu) in enumerate(zip(LAMBDA, MU), start=1):
        A_t = (
            mu / (lam + mu)
            + (lam / (lam + mu)) * np.exp(-(lam + mu) * time_grid)
        )

        plt.plot(time_grid, A_t, linewidth=1.7, label=f"C{i}")

    plt.xlabel("Time t (hours)", fontsize=12)
    plt.ylabel(r"Availability $A_i(t)$", fontsize=12)
    plt.title(
        "CTMC Time-Dependent Availability Curves for 15 Components",
        fontsize=15,
        fontweight="bold"
    )

    plt.ylim(0.93, 1.001)
    plt.xlim(0, T)
    plt.grid(True, alpha=0.3)
    plt.legend(ncol=3, fontsize=8)
    plt.tight_layout()

    plt.savefig(save_path, dpi=300, bbox_inches="tight")
    plt.show()
